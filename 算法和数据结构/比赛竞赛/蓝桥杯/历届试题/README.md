#include<iostream>
#include<stdio.h>

using namespace std;

int main() {
	int a, b, c;
	scanf("%d%d%d", &a, &b, &c);

		if( i % a == 0 && i % b == 0 && i % c == 0) {

//  最小公倍数=两数之积/这两个数的最大公约数
#include<iostream>
#include<stdio.h>

using namespace std;

// 获取最大公约数 
int gcd(int a, int b) {
	return !(a % b)? b:gcd(b, a%b);
} 

int main() {
	int a, b, c;
	scanf("%d%d%d", &a, &b, &c);
	
	//  最小公倍数=两数之积/这两个数的最大公约数
	// 按照公式应该这样写 a * b / gcd(a, b), 但可能会超过数据规模 
	int temp = a / gcd(a, b) * b;
	
	printf("%d", temp / gcd(temp, c) * c);
	
	return 0;
}
